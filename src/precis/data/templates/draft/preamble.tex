% precis draft export — standard preamble (ADR 0033 Tier-B).
%
% Checked into the repo and copied verbatim into every exported draft.
% The export engine assembles the rest of the document (title, acronym
% definitions, body, glossary, bibliography) around this block, so keep
% it package-setup only — no document content here.
%
% Compile with latexmk so biber (bibliography) and makeglossaries
% (glossary) both run:  latexmk main.tex  (lualatex, per latexmkrc).
\documentclass[11pt]{article}

% Engine is lualatex (UTF-8 native) — no inputenc. fontenc T1 + lmodern
% still give the Latin Modern look; unmapped Unicode degrades to a
% missing-glyph warning rather than a fatal error.
\usepackage[T1]{fontenc}
\usepackage{lmodern}

% CJK glyph coverage. Latin Modern has no CJK glyphs, so a Japanese / Chinese
% / Korean character (a paper title, a concept name) would otherwise render as
% a missing-glyph rule under lualatex. fontspec + \newfontfamily give us a CJK
% face and the exporter wraps detected CJK runs in \cjktext{...}
% (export/latex.py::_wrap_cjk). \cjktext defaults to identity, so a host with
% no CJK font still compiles (the glyph just stays missing) — it never throws
% an undefined-control-sequence error. HaranoAjiMincho ships with TeX Live
% (also the luatexja default); Noto Serif CJK is the common system fallback.
\usepackage{fontspec}
\newcommand{\cjktext}[1]{#1}
\IfFontExistsTF{HaranoAjiMincho}{%
  \newfontfamily\cjkfont{HaranoAjiMincho}%
  \renewcommand{\cjktext}[1]{{\cjkfont #1}}%
}{\IfFontExistsTF{Noto Serif CJK SC}{%
  \newfontfamily\cjkfont{Noto Serif CJK SC}%
  \renewcommand{\cjktext}[1]{{\cjkfont #1}}%
}{\IfFontExistsTF{Noto Serif CJK JP}{%
  \newfontfamily\cjkfont{Noto Serif CJK JP}%
  \renewcommand{\cjktext}[1]{{\cjkfont #1}}%
}{}}}

\usepackage[margin=1in]{geometry}
\usepackage{microtype}
\usepackage{amsmath,amssymb}
\usepackage{graphicx}
\usepackage{booktabs}
\usepackage{array}      % >{...}p{} column types for wrapping table cells
\usepackage{longtable}  % page-breaking data tables (export/latex.py)
\usepackage{pdfpages}   % \includepdf appendix for the referenced-sources bundle (include_sources)
\usepackage{listings}
\usepackage{csquotes}

% Bibliography — biblatex + biber. The export engine writes refs.bib
% from the cited paper refs (DOI included when known).
\usepackage[backend=biber,style=numeric-comp,sorting=none]{biblatex}

% Hyperlinks + clever cross-references (intra-draft ¶ refs become
% \cref{chunk:<handle>} pointing at \label{chunk:<handle>}).
\usepackage[hidelinks]{hyperref}
\usepackage{cleveref}

% Author byline + affiliations (export/latex.py build_author_block).
% authblk turns multiple \author + \affil lines into a numbered
% author↔affiliation block under \maketitle; a single shared
% affiliation renders unnumbered.
\usepackage{authblk}
\renewcommand\Affilfont{\small\itshape}

% Glossary / acronyms. `acronym` makes the acronym list; the long-short
% style means first use expands to the full term and every later use is
% the abbreviation. The page-number "where it occurs" list is on by
% default (we do NOT pass nonumberlist), per the export design.
\usepackage[acronym,toc]{glossaries-extra}
\setabbreviationstyle[acronym]{long-short}

% Screen-only abbreviation tooltips (pdfcomment, after hyperref/glossaries).
% First use of an acronym still expands inline; every LATER use is the bare
% short wrapped in a non-printing PDF tooltip that reveals the full term on
% hover — mirroring the web reader's instant popup without touching the
% printed page. Only later uses are wrapped: a \pdftooltip box can't break
% across a line and the first-use expansion is multi-word. hyper=false avoids
% overlapping the glossary link rectangle with the tooltip annotation.
% Loaded defensively: if pdfcomment is absent the macros degrade to a plain
% \gls / \glspl, so a draft still compiles (just without the hover). The
% \newcommand definitions live under a primitive \ifdefined conditional
% (branches delimited by \else/\fi, NOT absorbed as a macro argument) — a
% \newcommand nested inside \IfFileExists{}{...}{...} trips "Illegal
% parameter number" because LaTeX copies the branch through \def\reserved@a,
% where the bare #1 is no longer a legal parameter.
\IfFileExists{pdfcomment.sty}{\usepackage{pdfcomment}}{}
\ifdefined\pdftooltip
  \newcommand{\glstip}[1]{\pdftooltip{\gls[hyper=false]{#1}}{\glsentrylong{#1}}}
  \newcommand{\glspltip}[1]{\pdftooltip{\glspl[hyper=false]{#1}}{\glsentrylong{#1}}}
\else
  \newcommand{\glstip}[1]{\gls{#1}}
  \newcommand{\glspltip}[1]{\glspl{#1}}
\fi

% Code/listing blocks (chunk_kind = listing/code).
\lstset{basicstyle=\ttfamily\small,breaklines=true,frame=single,
        columns=fullflexible,keepspaces=true}

% Aside / box chunks render inside this lightweight framed environment.
\newenvironment{precisaside}{%
  \par\medskip\noindent\begin{quote}\itshape%
}{%
  \end{quote}\medskip%
}
